⚡ Optimized Layer Stack

1FROM node:20-alpine
2WORKDIR /usr/src/app
3COPY package*.json ./
4RUN npm ci
5RUN dd if=/dev/zero of=heavy-bundle.tar.gz bs=1M count=25 && rm heavy-bundle.tar.gz
